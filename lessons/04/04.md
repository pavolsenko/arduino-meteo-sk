void sensor() {
    float h = dht.readHumidity();
    float t = dht.readTemperature();

    while(isnan(h)) {
        dht.readHumidity();
    }

    while(isnan(t)) {
        dht.readTemperature();
    }

    lcd.setCursor(0, 0);
    lcd.print(t);

    lcd.setCursor(0, 1);
    lcd.print(h);
}

```C
#include <Wire.h>
#include <SPI.h>
#include <RTClib.h>
#include <LiquidCrystal.h>

RTC_DS1307 RTC; 
LiquidCrystal lcd(16, 14, 12, 13, 0, 2);

void setup() 
{
    lcd.begin(16, 2); 
    Wire.begin();
    RTC.begin();
    RTC.adjust(DateTime(__DATE__, __TIME__));
}

void loop() 
{
    DateTime now = RTC.now();

    lcd.setCursor(0, 0);

    if (now.day() < 10) {
        lcd.print('0');
        lcd.print(now.day(), DEC);
        lcd.print('.');
    } else {
        lcd.print(now.day(), DEC);
        lcd.print('.');
    }
    

    if (now.month() < 10) {
        lcd.print('0');
        lcd.print(now.month(), DEC);
        lcd.print('.');
    } else {
        lcd.print(now.month(), DEC);
        lcd.print('.');
    }
    
    lcd.print(now.year());
        
    lcd.setCursor(0, 1);

    if (now.hour() < 10) {
        lcd.print('0');
        lcd.print(now.hour());
    } else {
        lcd.print(now.hour());
    }

    lcd.print(':');

    if (now.minute() < 10) {
        lcd.print('0');
        lcd.print(now.minute());
    } else {
        lcd.print(now.minute());
        
    }

    lcd.print(':');

    if (now.second() < 10) {
        lcd.print('0');
        lcd.print(now.second());
    } else {
        lcd.print(now.second());
    }

    delay(400);
}
```
